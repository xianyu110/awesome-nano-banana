```javascript
// 主题状态管理
let currentTheme = 'dark'; // 默认黑暗模式

// 主题应用函数
function applyTheme() {
    const body = document.body;
    if (currentTheme === 'light') {
        body.classList.add('light-theme');
    } else {
        body.classList.remove('light-theme');
    }
}
```
